algoritmo
declare	alturas[10]
		menor , indice , j , i , menor numerico
alturas[9,8,7,6,1,2,4,3,5,3]
menor<-100
indice<-0

para j <-0 ate 10 faça
inicio
    para i <-j, len(alturas)):
        se alturas[i]<menor
		entao	inicio
				menor<-alturas[i]
				indice<-i
				fim
    alturas.insert(j, menor)
    escreva alturas[j]
    menor<-100
fim
para i <- 1 ate 10 faça
inicio
escreva alturas[i]
fim
fim_algoritmo
