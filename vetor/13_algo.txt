algoritmo
declare	candidatos[100] numerico
		tem , rodar logico
		voto literal
		n  , i,eleito , maior , eleito numerico
tem<-Falso
voto<-""
rodar<-Verdadeiro

leia n
eleito<-0
maior<-0

para i <-1 ate n faça
inicio
    candidatos[1]<-[leia, 0]
fim
enquanto rodar faça
inicio
    leia voto
    se voto=="fim":
	entao	rodar <- Falso
    senao
        para i <-0 ate n faça
		inicio
           se voto==candidatos[i][0]
           entao	candidatos[i][1]<-candidatos[i][1]+1
		fim
fim
para i <-0 ate n
inicio
    se candidatos[i][1]>maior
	inicio
       maior<-candidatos[i][1]
       eleito<-i
	fim
fim        
escreva candidatos[eleito][0]
fim_algoritmo
