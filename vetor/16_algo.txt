algoritmo
declare	questoes , gabarito , quantidade , aux , pontos , i , j , aux2 numerico
		alunos[quantidade], gabarito[questoes] numerico
questoes <- 30
leia quantidade 
aux<-0
pontos <- 0

#recebe gabarito
para i <-1 ate questoes faça
inicio
    leia gabarito[i]
fim
# recebe matricula, resostas 
para i <-1 ate quantidade faça # i - aluno
inicio
    leia aux
    alunos[i]<-aux
    para j <-0 ate questoesFalso # j - questoes
        leia aux2
        alunos[i].[j]<-aux2
        se gabarito[j]==aux2
		entao	pontos<- pontos+1
    alunos[i][j]<-pontos
    pontos<-0
fim    
#imprime resultados
para i <-0 ate quantidade faça
inicio
	escreva "matricula", alunos[i][0], " nota ->",  alunos[i][31])
fim
fim_algoritmo
