algoritmo
declare	 n , soma , variancia , desvio , i , media
		valores[n]
leia n
soma<-0
variancia <- 0
desvio <- 0

#   recebe os numeros e adiciona no somatorio
para i <-0 ate n
inicio
    leia numemro 
    valores[i]<-numemro
    soma<-soma+numemro
fim
media <- soma/n
para i <-0 ate n
inicio
    variancia<- variancia + (valores[i]-media)**2
fim
variancia <- variancia/(n-1)#    variancia
desvio <- variancia ** (1/2)#    desvio

escreva media
escreva  variancia
escreva  desvio
fim_algoritmo
