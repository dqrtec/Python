ALGORITMO
DECLARE n, maior, cont NUMERICO
n = 0
cont = 0
maior = 0
ESCREVA "*****Programa que imprime o maior número dentre os digitados*****"
ENQUANTO n >= 0 FAÇA
	ESCREVA "Digite um número negativo para sair do programa.\nDigite um número positivo:"
	LEIA n
	SE n >= 0 ENTAO
		INICIO
			SE n > maior ENTAO
				INICIO
					maior = n
				FIM
		FIM
	cont = cont + 1

SE cont == 0 ENTAO
	INICIO
		ESCREVA "Você não digitou nenhum número."
	FIM
SENAO
	INICIO			
		ESCREVA "O maior número é", maior
	FIM

FIM_ALGORITMO
